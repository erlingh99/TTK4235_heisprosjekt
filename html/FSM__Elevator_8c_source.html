13 void elevatorStateMachine()
14 {
15  switch (e->elevatorState)
16  {
17  case INIT:
18  hardware_command_movement(HARDWARE_MOVEMENT_DOWN);
19  if (e->floor >= 0)
20  {
21  e->elevatorState = IDLE;
22  fprintf(stderr, "STATE change: INIT->IDLE\n");
23  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
24  }
25  break;
26  case IDLE:
27  closeDoor(e);
28  if (e->doorState == CLOSED && hasOrders(e->orders))
29  {
30  e->elevatorState = MOVING;
31  fprintf(stderr, "STATE change: IDLE->MOVING\n");
32  }
33  break;
34  case MOVING:
35  { //curly braces to define scope of currOrder
36  int currOrder = destination(e->orders, e->floor, e->direction);
37  if (currOrder == -1)
38  exit(0);
39 
40  if ((e->floor == currOrder) && atFloor())
41  {
42  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
43  e->elevatorState = IDLE;
44  fprintf(stderr, "STATE change: MOVING->IDLE\n");
45  openDoor(e);
46  orderCompleted(e->orders, e->floor);
47 
48  for (int b = 0; b<HARDWARE_NUMBER_OF_BUTTONS; b++)
49  hardware_command_order_light(e->floor, b, 0);
50  }
51  else if (e->floor < currOrder)
52  {
53  hardware_command_movement(HARDWARE_MOVEMENT_UP);
54  e->direction = HARDWARE_MOVEMENT_UP;
55  }
56  else if (e->floor > currOrder)
57  {
58  hardware_command_movement(HARDWARE_MOVEMENT_DOWN);
59  e->direction = HARDWARE_MOVEMENT_DOWN;
60  }
61  else if (e->floor == currOrder && !e->moved)
62  {
63  //Stopped between floors and having order back to prev floor
64  if (e->direction == HARDWARE_MOVEMENT_UP)
65  {
66  hardware_command_movement(HARDWARE_MOVEMENT_DOWN); //opposite dir
67  e->direction = HARDWARE_MOVEMENT_DOWN;
68  }
69  else
70  {
71  hardware_command_movement(HARDWARE_MOVEMENT_UP);
72  e->direction = HARDWARE_MOVEMENT_UP;
73  }
74  }
75  e->moved = true;
76  break;
77  }
78  case STOPPED:
79  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
80  openDoor(e); //keeps the door open if the elevator is stopped at a floor
81  e->moved = false;
82  break;
83  }
