14 void elevatorStateMachine()
15 {
16  switch (p_elevator->elevatorState)
17  {
18  case INIT:
19  hardware_command_movement(HARDWARE_MOVEMENT_DOWN);
20  if (atFloor())
21  {
22  p_elevator->elevatorState = IDLE;
23  fprintf(stderr, "STATE change: INIT->IDLE\n");
24  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
25  }
26  break;
27  case IDLE:
28  closeDoor(p_elevator);
29  if (p_elevator->doorState == CLOSED && hasOrders(p_elevator->orders))
30  {
31  p_elevator->elevatorState = MOVING;
32  fprintf(stderr, "STATE change: IDLE->MOVING\n");
33  }
34  break;
35  case MOVING:
36  { //curly braces to define scope of currOrder
37  int currOrder = findDestination(p_elevator->orders, p_elevator->floor, p_elevator->direction);
38  if (currOrder < 0 || currOrder >= HARDWARE_NUMBER_OF_FLOORS)
39  exit(0);
40 
41  if ((p_elevator->floor == currOrder) && atFloor())
42  {
43  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
44  p_elevator->elevatorState = IDLE;
45  fprintf(stderr, "STATE change: MOVING->IDLE\n");
46  openDoor(p_elevator);
47  orderCompleted(p_elevator->orders, p_elevator->floor);
48 
49  for (int b = 0; b<HARDWARE_NUMBER_OF_BUTTONS; b++)
50  hardware_command_order_light(p_elevator->floor, b, 0);
51  }
52  else if (p_elevator->floor < currOrder)
53  {
54  hardware_command_movement(HARDWARE_MOVEMENT_UP);
55  p_elevator->direction = HARDWARE_MOVEMENT_UP;
56  }
57  else if (p_elevator->floor > currOrder)
58  {
59  hardware_command_movement(HARDWARE_MOVEMENT_DOWN);
60  p_elevator->direction = HARDWARE_MOVEMENT_DOWN;
61  }
62  else if (p_elevator->floor == currOrder && !p_elevator->moved)
63  {
64  //Stopped between floors and having order back to prev floor
65  if (p_elevator->direction == HARDWARE_MOVEMENT_UP)
66  {
67  hardware_command_movement(HARDWARE_MOVEMENT_DOWN); //opposite dir
68  p_elevator->direction = HARDWARE_MOVEMENT_DOWN;
69  }
70  else
71  {
72  hardware_command_movement(HARDWARE_MOVEMENT_UP);
73  p_elevator->direction = HARDWARE_MOVEMENT_UP;
74  }
75  }
76  p_elevator->moved = true;
77  break;
78  }
79  case STOPPED:
80  hardware_command_movement(HARDWARE_MOVEMENT_STOP);
81  openDoor(p_elevator); //keeps the door open if the elevator is stopped at a floor
82  p_elevator->moved = false;
83  break;
84  }
85 }
